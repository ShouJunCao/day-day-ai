一、什么是Token？

Token是大模型处理文本的最小单位。它不是字符，也不是单词，而是由tokenizer（分词器）将文本切分后得到的片段。

1.1 Token与字符的关系

文本类型示例字符数Token数比率
英文短句"Hello, how are you?"2163.5字符/token
中文短句"你好，最近怎么样？"971.3字符/token
代码"def hello():"1352.6字符/token
混合文本"Python是一门好语言"1281.5字符/token
⚠️ 重要：中文文本的token密度比英文高得多。同样含义的内容，中文大约消耗1.5-2倍的token。这意味着面向中文用户的应用，API成本通常更高。

1.2 输入 vs 输出定价

大模型API的一个重要特点：输出token的价格通常是输入的3-5倍。这是因为生成文本比理解文本需要更多计算资源。

模型输入价格（/百万token）输出价格（/百万token）输入:输出比
GPT-4o$2.50$10.001:4
GPT-4o-mini$0.15$0.601:4
DeepSeek-Chat$0.27$1.101:4
Qwen-Turbo$0.10$0.301:3
Claude Sonnet 4$3.00$15.001:5
💡 省钱技巧：控制输出长度比控制输入长度更有效。设置合理的 max_tokens 参数，避免模型"话痨"。

1.3 实际费用计算案例

假设你开发了一个AI客服机器人，每天处理1000次对话，每次对话平均消耗800个输入token和300个输出token：

二、精确计算Token数量

要控制成本，首先得知道每次调用消耗了多少token。OpenAI开源的tiktoken库是目前最准确的token计数工具。

2.1 安装与基本用法

📄 Token计数基础
# 安装：pip install tiktoken
import tiktoken

# 获取指定模型的tokenizer
# GPT-4o 使用 o200k_base 编码
# GPT-4/GPT-3.5 使用 cl100k_base 编码
encoder = tiktoken.get_encoding("o200k_base")

# 编码文本 → token列表
tokens = encoder.encode("Hello, how are you?")
print(f"Token数: {len(tokens)}")  # 输出: 6

# 中文文本的token数更多
cn_tokens = encoder.encode("你好，最近怎么样？")
print(f"中文Token数: {len(cn_tokens)}")  # 输出: 约7-9

2.2 封装为可复用的计数器

📄 token_counter.py
import tiktoken
from typing import Dict, List


class TokenCounter:
    """Token计数器 - 支持多模型的token计数

    封装tiktoken库，提供单条文本和消息列表的token计数。
    使用缓存避免重复加载tokenizer（加载较慢）。
    """

    # 模型到tokenizer编码的映射
    MODEL_ENCODINGS: Dict[str, str] = {
        "gpt-4o": "o200k_base",
        "gpt-4o-mini": "o200k_base",
        "gpt-4": "cl100k_base",
        "gpt-3.5-turbo": "cl100k_base",
        "deepseek-chat": "cl100k_base",
    }

    def __init__(self):
        self.encoders: Dict[str, tiktoken.Encoding] = {}

    def _get_encoder(self, model: str) -> tiktoken.Encoding:
        """获取或创建tokenizer（带缓存）"""
        if model not in self.encoders:
            enc_name = self.MODEL_ENCODINGS.get(model, "cl100k_base")
            self.encoders[model] = tiktoken.get_encoding(enc_name)
        return self.encoders[model]

    def count_tokens(self, text: str, model: str = "gpt-4o-mini") -> int:
        """计算文本的token数量"""
        encoder = self._get_encoder(model)
        return len(encoder.encode(text))

    def count_messages(
        self,
        messages: List[Dict[str, str]],
        model: str = "gpt-4o-mini",
    ) -> int:
        """计算消息列表的token数（含格式开销）

        每条消息额外消耗约3个token（role标记+分隔符）。
        """
        encoder = self._get_encoder(model)
        total = 0

        for msg in messages:
            total += 3  # 消息格式开销
            total += len(encoder.encode(msg["role"]))
            total += len(encoder.encode(msg["content"]))

        total += 3  # 回复预填充
        return total

from dataclasses import dataclass
from typing import Dict, List


@dataclass
class ModelPricing:
    """模型定价信息"""
    model_name: str
    input_price: float   # 每百万token，美元
    output_price: float  # 每百万token，美元
    provider: str = ""


class CostCalculator:
    """API费用计算器 - 预估和追踪调用成本"""

    # 模型定价表（2025年最新价格）
    PRICING: Dict[str, ModelPricing] = {
        "gpt-4o": ModelPricing(
            "gpt-4o", 2.50, 10.00, "OpenAI"
        ),
        "gpt-4o-mini": ModelPricing(
            "gpt-4o-mini", 0.15, 0.60, "OpenAI"
        ),
        "deepseek-chat": ModelPricing(
            "deepseek-chat", 0.27, 1.10, "DeepSeek"
        ),
    }

    def __init__(self):
        self.usage_history: List[dict] = []

    def estimate_cost(
        self,
        model: str,
        input_tokens: int,
        output_tokens: int,
    ) -> float:
        """预估单次调用费用（美元）"""
        pricing = self.PRICING[model]
        input_cost = (input_tokens / 1_000_000) * pricing.input_price
        output_cost = (output_tokens / 1_000_000) * pricing.output_price
        return input_cost + output_cost

    def compare_models(
        self,
        input_tokens: int,
        output_tokens: int,
    ) -> List[dict]:
        """多模型费用对比（按费用排序）"""
        results = []
        for name, pricing in self.PRICING.items():
            cost = self.estimate_cost(name, input_tokens, output_tokens)
            results.append({
                "model": name,
                "provider": pricing.provider,
                "cost": cost,
            })
        return sorted(results, key=lambda x: x["cost"])

四、预算控制器：防止费用失控

在生产环境中，API费用可能因以下原因失控：恶意用户发送超长prompt、代码bug导致无限循环调用、并发量激增超出预期。我们需要多层预算控制机制。

4.1 多层预算控制策略

控制层级目的典型限制
单次请求防止超长prompt最大4000 token/次
用户级别防止单用户滥用$5/天、$50/月
全局级别控制总成本$100/天
速率限制防止恶意刷接口60次/分钟

4.2 预算控制器实现

📄 budget_controller.py
from dataclasses import dataclass
from typing import Dict, Optional
from collections import defaultdict
import time


@dataclass
class BudgetConfig:
    """预算配置"""
    max_tokens_per_request: int = 4000
    daily_budget_per_user: float = 5.0      # 美元
    monthly_budget_per_user: float = 50.0    # 美元
    global_daily_budget: float = 100.0      # 美元
    max_requests_per_minute: int = 60


class BudgetExceededError(Exception):
    """预算超限异常"""
    def __init__(self, reason: str, current: float, limit: float):
        self.reason = reason
        super().__init__(f"预算超限: {reason} (当前: ${current:.4f}, 限额: ${limit:.4f})")


class BudgetController:
    """预算控制器 - 多层预算检查

    在每次API调用前检查预算状态，超限时抛出异常阻止调用。
    """

    def __init__(self, config: Optional[BudgetConfig] = None):
        self.config = config or BudgetConfig()
        self.daily_usage: Dict[str, float] = defaultdict(float)
        self.global_daily_usage: float = 0.0
        self.request_timestamps: Dict[str, list] = defaultdict(list)

    def check_budget(self, user_id: str, estimated_cost: float) -> None:
        """调用前检查预算是否充足"""
        # 检查用户每日预算
        daily_used = self.daily_usage[user_id]
        if daily_used + estimated_cost > self.config.daily_budget_per_user:
            raise BudgetExceededError(
                f"用户 {user_id} 每日预算超限",
                daily_used, self.config.daily_budget_per_user
            )

        # 检查全局预算
        if self.global_daily_usage + estimated_cost > self.config.global_daily_budget:
            raise BudgetExceededError(
                "系统全局每日预算超限",
                self.global_daily_usage, self.config.global_daily_budget
            )

    def record_cost(self, user_id: str, cost: float) -> None:
        """记录实际消费"""
        self.daily_usage[user_id] += cost
        self.global_daily_usage += cost

    def check_rate_limit(self, user_id: str) -> None:
        """检查请求频率限制（滑动窗口算法）"""
        now = time.time()
        window_start = now - 60  # 60秒窗口

        # 清理过期记录
        self.request_timestamps[user_id] = [
            ts for ts in self.request_timestamps[user_id]
            if ts > window_start
        ]

        current_count = len(self.request_timestamps[user_id])
        if current_count >= self.config.max_requests_per_minute:
            raise BudgetExceededError(
                "请求频率超限", current_count,
                self.config.max_requests_per_minute
            )

        self.request_timestamps[user_id].append(now)

五、Prompt优化：减少不必要的Token消耗

除了选择便宜的模型，优化prompt本身也能大幅降低成本。常见浪费包括：过长的system prompt、重复的历史消息、冗余的few-shot示例等。

5.1 对话历史裁剪

多轮对话时，历史消息会不断累积，每次都会计费。需要在token预算内保留最近的消息，裁剪掉最早的历史。

📄 对话历史裁剪策略
from typing import List, Dict, Callable


class PromptOptimizer:
    """Prompt优化工具 - 减少token消耗

    核心策略：
    1. 始终保留system消息
    2. 从最新消息开始保留
    3. 超出token预算时裁剪最早的消息
    """

    def __init__(self, max_context_tokens: int = 4000):
        self.max_context_tokens = max_context_tokens

    def trim_history(
        self,
        messages: List[Dict[str, str]],
        token_counter: Callable[[str], int],
        reserved_for_output: int = 500,
    ) -> List[Dict[str, str]]:
        """裁剪对话历史，保留最新消息"""
        available = self.max_context_tokens - reserved_for_output

        # 始终保留system消息
        system_msg = None
        if messages and messages[0]["role"] == "system":
            system_msg = messages[0]
            available -= token_counter(system_msg["content"])
            messages = messages[1:]

        # 从最新消息开始，向前保留
        kept = []
        used = 0
        for msg in reversed(messages):
            msg_tokens = token_counter(msg["content"]) + 3
            if used + msg_tokens > available:
                break
            kept.insert(0, msg)
            used += msg_tokens

        return ([system_msg] + kept) if system_msg else kept

5.3 成本控制决策树

在实际项目中，面对不同的需求场景，如何选择合适的成本控制策略？

场景日调用量推荐策略预计月费
内部工具（低并发）<100次选便宜模型 + 基础预算<$5
C端产品（中并发）1000-10000次小模型为主 + 历史裁剪 + 用户预算$50-200
企业级服务（高并发）>10000次模型路由 + 缓存 + 全局预算 + 告警$200-2000+
批处理任务按需最便宜模型 + max_tokens限制按量
💡 经验法则：80%的日常对话任务可以用GPT-4o-mini或DeepSeek-Chat完成，不需要用GPT-4o。先用小模型，效果不够再升级。

六、用量监控与告警

生产环境需要持续监控API使用情况，及时发现异常（费用突然飙升、某用户消费远超其他人、接近预算上限等）。

6.1 监控系统核心功能

功能说明实现方式
实时记录每次调用都记录用量在API调用后记录
多维聚合按用户、模型、时间统计内存聚合或数据库查询
阈值告警超过阈值时触发通知回调函数 + 冷却时间
报告导出生成用量报告JSON/CSV格式

from typing import Dict, List, Callable
from collections import defaultdict
from dataclasses import dataclass
import time


class UsageMonitor:
    """用量监控器 - 实时追踪API消费

    支持多维度统计聚合和自定义告警规则。
    """

    def __init__(self):
        self.cost_by_user: Dict[str, float] = defaultdict(float)
        self.cost_by_model: Dict[str, float] = defaultdict(float)
        self.total_cost: float = 0.0
        self.total_requests: int = 0
        self.alerts: List[dict] = []

    def record(
        self,
        user_id: str,
        model: str,
        input_tokens: int,
        output_tokens: int,
        cost: float,
    ) -> None:
        """记录一次API调用"""
        self.cost_by_user[user_id] += cost
        self.cost_by_model[model] += cost
        self.total_cost += cost
        self.total_requests += 1

        # 检查告警
        self._check_alerts()

    def add_alert(
        self,
        name: str,
        threshold: float,
        callback: Callable,
    ) -> None:
        """添加告警规则"""
        self.alerts.append({
            "name": name,
            "threshold": threshold,
            "callback": callback,
            "last_triggered": 0,
        })

    def _check_alerts(self) -> None:
        """检查是否触发告警"""
        now = time.time()
        for alert in self.alerts:
            # 冷却时间：5分钟
            if now - alert["last_triggered"] < 300:
                continue

            if self.total_cost > alert["threshold"]:
                alert["callback"](
                    f"⚠️ {alert['name']}: 总费用${self.total_cost:.4f}"
                )
                alert["last_triggered"] = now

    def get_summary(self) -> dict:
        """获取用量摘要"""
        return {
            "total_requests": self.total_requests,
            "total_cost": round(self.total_cost, 6),
            "cost_by_user": dict(self.cost_by_user),
            "cost_by_model": dict(self.cost_by_model),
            "avg_cost_per_request": round(
                self.total_cost / max(self.total_requests, 1), 6
            ),
        }
