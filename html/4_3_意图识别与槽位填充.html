4.3 意图识别与槽位填充

意图分类 (Intent Classification) · 槽位提取 (Slot Filling) · 语义解析 (Semantic Parsing)

📌 核心要点：意图识别与槽位填充是对话系统的核心理解层。意图识别回答"用户想做什么"，槽位填充回答"用户提供了哪些关键参数"。传统方法依赖分类模型+序列标注，现代方法可以直接用 LLM 一次完成。

什么是对话理解？

当用户说"帮我订明天北京到上海的机票"时，对话系统需要解析出：

  • 意图 (Intent)：订机票
  • 槽位 (Slots)：出发地=北京, 目的地=上海, 时间=明天

传统方法 vs LLM 方法

在 LLM 出现之前，意图识别和槽位填充通常由两个独立的模型串联完成：

  1. 意图分类器：将用户语句分类到预定义的意图类别（如 BERT + 全连接层）
  2. 槽位标注器：使用 BIO/BILOU 标记法进行序列标注（如 BiLSTM-CRF）
⚠️ 传统方法的问题：需要大量标注数据、无法处理未见过的意图、模型之间错误累积、维护成本高。LLM 的出现将这些问题一并解决。

实战代码 1：传统方法（BERT 意图分类）

# intent_classifier.py — 基于 BERT 的意图分类
import torch
from transformers import BertTokenizer, BertForSequenceClassification

class IntentClassifier:
    """使用预训练 BERT 进行意图分类"""

    INTENT_LABELS = [
        "greet", "goodbye", "inform", "request",
        "book_flight", "query_weather", "query_restaurant"
    ]

    def __init__(self, model_name: str = "bert-base-chinese"):
        self.tokenizer = BertTokenizer.from_pretrained(model_name)
        self.model = BertForSequenceClassification.from_pretrained(
            model_name, num_labels=len(self.INTENT_LABELS)
        )
        self.model.eval()

    def predict(self, text: str) -> dict:
        """预测输入文本的意图"""
        inputs = self.tokenizer(
            text, return_tensors="pt", truncation=True, max_length=128
        )
        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.softmax(outputs.logits, dim=-1)
            top_idx = probs.argmax().item()

        return {
            "intent": self.INTENT_LABELS[top_idx],
            "confidence": probs[0][top_idx].item(),
            "all_scores": {
                label: probs[0][i].item()
                for i, label in enumerate(self.INTENT_LABELS)
            }
        }

# 使用示例
classifier = IntentClassifier()
result = classifier.predict("帮我订明天北京到上海的机票")
print(f"意图: {result['intent']} (置信度: {result['confidence']:.2%})")

实战代码 2：传统方法（CRF 槽位标注）

# slot_filler.py — 基于 CRF 的槽位填充
from sklearn_crfsuite import CRF
from typing import List

class SlotFiller:
    """使用 CRF 进行序列标注（BIO 标记法）"""

    def __init__(self):
        self.crf = CRF(
            algorithm='lbfgs',
            c1=0.1,
            c2=0.01,
            max_iterations=100,
            all_possible_transitions=True
        )

    def _word2features(self, sentence: List[str], i: int) -> dict:
        """提取单个词的特征"""
        word = sentence[i]
        return {
            'word': word,
            'is_title': word.istitle(),
            'is_digit': word.isdigit(),
            'prefix2': word[:2],
            'suffix2': word[-2:],
            'pos': self._get_pos(word),
            'bias': 1.0,
        }

    def _get_pos(self, word: str) -> str:
        """简易词性标注（生产环境用 HanLP/jieba）"""
        if any(c.isdigit() for c in word):
            return "NUM"
        time_words = {"明天", "今天", "后天", "早上", "晚上"}
        place_words = {"北京", "上海", "广州", "深圳", "成都"}
        if word in time_words:
            return "TIME"
        if word in place_words:
            return "PLACE"
        return "OTHER"

    def _sent2features(self, sentence: List[str]) -> List[dict]:
        """提取整句的特征序列"""
        return [self._word2features(sentence, i) for i in range(len(sentence))]

    def fit(self, X: List[List[str]], y: List[List[str]]) -> None:
        """训练 CRF 模型"""
        X_feats = [self._sent2features(s) for s in X]
        self.crf.fit(X_feats, y)

    def predict(self, sentence: List[str]) -> List[str]:
        """预测槽位标签序列"""
        X_feats = self._sent2features(sentence)
        return self.crf.predict([X_feats])[0]

# B-departure I-departure B-destination I-destination
# 北京      -           上海          -
# B-date I-date
# 明天      -

# llm_nlu.py — 使用 LLM + Function Calling 进行对话理解
import os
import json
from typing import Literal, Optional
from pydantic import BaseModel, Field

# --- Schema 定义 ---
class FlightSlots(BaseModel):
    """订机票的槽位"""
    departure: str = Field(description="出发城市")
    arrival: str = Field(description="到达城市")
    date: str = Field(description="出发日期，格式 YYYY-MM-DD")
    return_date: Optional[str] = Field(
        default=None, description="返回日期（单程时省略）"
    )
    cabin_class: Optional[str] = Field(
        default=None, description="舱位等级"
    )

class NLUResult(BaseModel):
    """对话理解结果"""
    intent: str = Field(description="意图名称")
    confidence: float = Field(description="置信度 0-1")
    slots: dict = Field(description="提取的槽位参数")
    missing_slots: list[str] = Field(
        description="缺失的必要槽位，用于后续追问"
    )

class DialogueUnderstandingClient:
    """
    基于 LLM 的对话理解客户端

    使用 Function Calling + Pydantic 确保输出格式
    """
    def __init__(self):
        from openai import OpenAI
        self.client = OpenAI(
            api_key=os.getenv("API_KEY"),
            base_url=os.getenv("BASE_URL", "https://api.openai.com/v1"),
        )
        self.model = os.getenv("MODEL", "gpt-4o")

    def understand(self, user_input: str) -> NLUResult:
        """解析用户输入的意图和槽位"""
        # System Prompt 中定义理解规则
        system_msg = {
            "role": "system",
            "content": (
                "你是一个对话理解专家。请分析用户输入，提取意图和槽位。\n"
                "支持的意图包括：book_flight, query_weather, "
                "query_restaurant, chat, unknown\n"
                "输出必须符合 NLUResult 格式。"
            )
        }

        response = self.client.beta.chat.completions.parse(
            model=self.model,
            messages=[system_msg, {"role": "user", "content": user_input}],
            response_format=NLUResult,
        )
        return response.choices[0].message.parsed

# 使用示例
client = DialogueUnderstandingClient()
result = client.understand("帮我订后天上海飞北京的商务舱")
print(f"意图: {result.intent}")
print(f"槽位: {result.slots}")
if result.missing_slots:
    print(f"缺失槽位: {result.missing_slots}")

实战代码 4：LLM 方法（JSON Schema 强制输出）

# llm_schema_nlu.py — 基于 JSON Schema 的槽位填充
import os
import json
import httpx
from typing import Optional

# 定义槽位 Schema（不依赖 Pydantic/OpenAI SDK）
FLIGHT_SCHEMA = {
    "type": "object",
    "properties": {
        "intent": {"type": "string"},
        "slots": {
            "type": "object",
            "properties": {
                "departure": {"type": "string"},
                "arrival": {"type": "string"},
                "date": {"type": "string"},
            },
            "required": ["departure", "arrival", "date"],
        },
    },
    "required": ["intent", "slots"],
}

class SchemaBasedNLU:
    """
    基于 JSON Schema 的对话理解
    适用于任何支持 JSON mode 的 LLM
    """

    def __init__(self):
        self.api_key = os.getenv("API_KEY", "")
        self.base_url = os.getenv("BASE_URL", "https://api.openai.com/v1")
        self.model = os.getenv("MODEL", "gpt-4o-mini")

    async def extract(self, text: str) -> Optional[dict]:
        """提取意图和槽位"""
        prompt = (
            f"请从以下文本中提取意图和槽位信息。\n\n"
            f"用户输入：{text}\n\n"
            f"请以以下 JSON 格式输出：\n"
            f"{json.dumps(FLIGHT_SCHEMA, indent=2, ensure_ascii=False)}"
        )

        async with httpx.AsyncClient() as client:
            resp = await client.post(
                f"{self.base_url}/chat/completions",
                json={
                    "model": self.model,
                    "messages": [{"role": "user", "content": prompt}],
                    "response_format": {"type": "json_object"},
                },
                headers={"Authorization": f"Bearer {self.api_key}"},
            )
            return resp.json()["choices"][0]["message"]["content"]

    def validate(self, result: dict) -> tuple[bool, list[str]]:
        """验证提取结果是否完整"""
        errors = []
        if not result.get("intent"):
            errors.append("缺少 intent")
        slots = result.get("slots", {})
        for field in ["departure", "arrival", "date"]:
            if field not in slots:
                errors.append(f"缺少必要槽位: {field}")
        return len(errors) == 0, errors

混合方案：分类器 + LLM

在生产环境中，为了平衡成本和效果，常采用混合方案：

  1. 先用轻量分类器（如 FastText / ONNX 模型）做意图初筛，成本极低
  2. 对高置信度结果直接提取槽位（规则匹配或轻量 NER）
  3. 对低置信度或复杂场景，回退到 LLM进行深度理解
💡 实战技巧：始终检查 LLM 输出中的 missing_slots，将其用于生成追问话术："好的，您想订后天上海飞北京的商务舱。请问是单程还是往返？"
